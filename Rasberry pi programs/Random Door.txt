import random
import Tkinter

window = Tkinter.Tk()

window.title("Random")
#set window size
window.geometry("300x300")

#Generate a random number
doorToGuess = random.randint(0,2)

def checkDoor(number):
        global doorToGuess
        if number == doorToGuess:
                lbl.configure(text = "Yes")
        else:
                lbl.configure(text = "no")

lbl = Tkinter.Label(window, text="Wich door contains the prize?")
lbl.pack()

#make 3 buttons
for i in range(3):
        btn= Tkinter.Button(text="Door" + str(i), command=lambda doorNo=i :  checkDoor(doorNo))
        btn.pack(side=Tkinter.LEFT)

window.mainloop()
